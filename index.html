Языки бывают мономорфными...

... и полиморфными.

                
                    $value = 'php';
                
            
                
                    // $value is string
                    print($value);
                
            
                
                    // $value is mixed
                    var_dump($value);
                
            
                
                    function php(): void {}

                    // $value is callable
                    'php'();
                    $value();
                
            
                
                    // $value is non-empty-string
                    // $value is literal-string
                    // $value is 'php'
                    // $value is lower-case-string
                    // $value is non-empty-lower-case-string
                    // $value is non-falsy-string
                    // $value is truthy-string
                    $value = 'php';
